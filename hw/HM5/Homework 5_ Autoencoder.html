HM5: Build a Supervised Autoencoder.¶

Name: Andrew Chen¶

PCA and the standard autoencoder are unsupervised dimensionality reduction methods, and their learned features are not discriminative. If you build a classifier upon the low-dimenional features extracted by PCA and autoencoder, you will find the classification accuracy very poor.

Linear discriminant analysis (LDA) is a traditionally supervised dimensionality reduction method for learning low-dimensional features which are highly discriminative. Likewise, can we extend autoencoder to supervised learning?

You are required to build and train a supervised autoencoder look like the following. You are required to add other layers properly to alleviate overfitting.

import numpy

def to_one_hot(y, num_class=10):
    results = numpy.zeros((len(y), num_class))
    for i, label in enumerate(y):
        results[i, label] = 1.
    return results

from keras.layers import Dense, Input, Dropout, Activation, BatchNormalization
from keras import models

input_img = Input(shape=(784,), name='input_img')

encode = Dense(256, name='encode1')(input_img)
encode = BatchNormalization(name='bn1')(encode)
encode = Activation("elu", name="act1")(encode)

encode = Dense(64, name='encode2')(encode)
encode = BatchNormalization(name='bn2')(encode)
encode = Activation("elu", name="act2")(encode)

encode = Dense(32, name='encode3')(encode)
encode = BatchNormalization(name='bn3')(encode)
encode = Activation("elu", name="act3")(encode)

encode = Dense(2, name='bottleneck')(encode)
encode = BatchNormalization(name='bn4')(encode)
encode = Activation("elu", name="act4")(encode)

decode = Dense(32, name='decode1')(encode)
decode = BatchNormalization(name='bn5')(decode)
decode = Activation("elu", name="act5")(decode)

decode = Dense(64, name='decode2')(decode)
decode = BatchNormalization(name='bn6')(decode)
decode = Activation("elu", name="act6")(decode)

decode = Dense(256, name='decode3')(decode)
decode = BatchNormalization(name='bn7')(decode)
decode = Activation("elu", name="act7")(decode)

ae = models.Model(input_img, decode)

# build the supervised autoencoder network
from keras.layers import Dense, Input, Activation, BatchNormalization
from keras import models

input_img = Input(shape=(784,), name='input_img')

# encoder network
encode = Dense(256, name='e1')(input_img)
encode = BatchNormalization()(encode)
encode = Activation("relu")(encode)

encode = Dense(128, name='e2')(encode)
encode = BatchNormalization()(encode)
encode = Activation("relu")(encode)

encode = Dense(32, name='e3')(encode)
encode = BatchNormalization()(encode)
encode = Activation("relu")(encode)

# bottleneck
bottleneck = Dense(2, name='bottleneck')(encode)
bottleneck = BatchNormalization()(bottleneck)
bottleneck = Activation("relu")(bottleneck)

# decoder network
decode1 = Dense(32, name='d1')(bottleneck)
decode1 = BatchNormalization()(decode1)
decode1 = Activation("relu")(decode1)

decode2 = Dense(128, name='d2')(decode1)
decode2 = BatchNormalization()(decode2)
decode2 = Activation("relu")(decode2)

decode3 = Dense(256, name='d3')(decode2)
decode3 = BatchNormalization()(decode3)
decode3 = Activation("relu")(decode3)

decode4 = Dense(784, name='d4')(decode3)
decode4 = BatchNormalization()(decode4)
decode4 = Activation("relu")(decode4)

# build a classifier upon the bottleneck layer
classifier1 = Dense(256, name='c1')(bottleneck)
classifier1 = BatchNormalization()(classifier1)
classifier1 = Activation("relu")(classifier1)

classifier2 = Dense(64, name='c2')(classifier1)
classifier2 = BatchNormalization()(classifier2)
classifier2 = Activation("relu")(classifier2)

classifier3 = Dense(10, name='c3')(classifier2)
classifier3 = BatchNormalization()(classifier3)
classifier3 = Activation("softmax")(classifier3)
